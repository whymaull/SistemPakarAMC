  <script>
    fetch("https://gg0l3mpr-3000.asse.devtunnels.ms//api/riwayat")
      .then(res => res.json())
      .then(data => {
        const counts = {};
        data.forEach(item => {
          if (!counts[item.diagnosis]) counts[item.diagnosis] = 0;
          counts[item.diagnosis]++;
        });

        const labels = Object.keys(counts);
        const values = Object.values(counts);

        new Chart(document.getElementById("chartDiagnosis"), {
          type: "bar",
          data: {
            labels: labels,
            datasets: [{
              label: "Jumlah Diagnosis",
              data: values,
              backgroundColor: "#f87171",
              borderRadius: 5,
            }]
          },
          options: {
            responsive: true,
            scales: {
              y: { beginAtZero: true }
            }
          }
        });
      });
  </script>
